# Netmanager 命令教程
   - nmcli device wifi list # 查看 wifi 列出附近的WIFI清单
   - nmcli device wifi connect SSID_OR_BSSID password PASSWORD # 连接到无线网络
   - nmcli device wifi connect SSID_OR_BSSID password PASSWORD hidden yes # 连接到无线网络
   - nmcli device wifi connect SSID_OR_BSSID password PASSWORD ifname DEVICE PROFILE_NAME
   - nmcli device disconnect ifname DEVICE # 断开连接
   - nmcli connection show # 获取连接信息
   - nmcli connection up NAME_OR_UUID # 激活一个连接
   - nmcli connection delete NAME_OR_UUID # 删除一个连接
   - nmcli device # 查看网络设备状态的清单
   - nmcli radio wifi off # 关闭wifi
