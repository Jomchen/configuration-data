alt + /                  # 内容辅助，补全功能
ctrl + e                 # 显示当前打开的所有文件
ctrl + d                 # 复制当前行到下一行
ctrl + w                 # 关闭当前标签
ctrl + shift + w         # 关闭打开的所有标签
ctrl + shift + b         # 启用或停止面包屑功能（在顶部显示，其实就是当前文件所在目录情况）
ctrl + shift + p         # 从括号对之前相互跳转（列如当前光标在左括号，按键后可以跳转到对应的右括号）
ctrl + l                 # 跳转到指定行号
ctrl + shift + l         # 在当前文件中搜索内容
ctrl + enter             # 在当前行下新启一行并编辑
ctrl + shift + enter     # 在当前行上新启一行并编辑
alt + up/down            # 光标所在行往上/下移动
alt + shift + left/right # 选中当前光标所在单词
alt + left/right         # 标签的左右切换
ctrl + shift + t         # 查找类名
ctrl + shift + r         # 全局查找资源
ctrl + h                 # 高级搜索
ctrl + t                 # 显示光标所在类的继承和实现结构
alt + shift + s          # 调出代码生成视图（一般用于生成 get/set, toString, hash 等方法）
ctrl + f7                # 工作区，导航区，等区域的切换
ctrl + shift + g         # 查询当前光标类或方法被引用的地方
ctrl + shift y/x         # 变小写/大写
