* 基本快捷键
   ctrl + t                # 查找类
   ctrl + shift + e        # 焦点跳转到当前文件所在目录

   ctrl + 0                # 焦点聚集于主侧栏
   ctrl + space            # 建议提示
   ctrl + shift + space    # 参数提示
   ctrl + .                # 快速修复
   shift + alt + o         # 调整 import 语句（替代 ctrl + . 不能修复导包的问题）
   shift + alt + u         # 重新加载项目（修改pom后有效果）
   shift + alt + .         # 自动修复
   shift + alt + <         # 自动修复（但是无效果）
   ctrl + b                # 显示/隐藏边缘按钮显示
   ctrl + shift + e        # show explorer / toggle focus
   ctrl + shift + f        # 搜索
   ctrl + shift + g        # 版本控制窗口
   ctrl + shift + d        # debug窗口
   ctrl + shift + x        # 扩展窗口
   ctrl + shift + h        # 替换文件
   ctrl + k ctrl + h       # 显示输出面板
   ctrl + p                # 到文件
   f8                      # 移动到下一个 error 或 warning 处
   shift + f8              # 移动到上一个 error 或 warning 处
   ctrl + `                # 调出终端
   ctrl + alt + -          # go back
   ctrl + alt + +          # go forward

* 常用定位
   f12                # 转到定义
   shift + f12        # 引用：转到引用
   shift + alt + f12  # 引用：查找所有引用
   ctrl + shift + f10 # 快速预览定义
   
   ctrl + f12         # 转到实现
   ctrl + shift + f12 # 查看实现（预览实现）
   ctrl + shift + r   # 编辑器：再次搜索
   ctrl + shift + f   # 搜索工作区

* debug
   f9                      # 切换断点
   f5                      # 开始 / 继续
   f11 / shift + f11       # 单步进入/出
   f10                     # step over
   shift + f5              # 停止
